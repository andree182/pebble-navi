cmake_minimum_required(VERSION 3.20)
project(pebble-navi C)

set(CMAKE_C_STANDARD 99)

# Tell CMake to re-run configuration when package.json changes
set_property(DIRECTORY APPEND PROPERTY CMAKE_CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/package.json")

# Read and parse message keys from package.json
if(EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/package.json")
    file(READ "${CMAKE_CURRENT_SOURCE_DIR}/package.json" PACKAGE_JSON_CONTENT)
    
    # Check if pebble.messageKeys exists
    string(JSON MESSAGE_KEYS_LEN ERROR_VARIABLE JSON_ERR LENGTH "${PACKAGE_JSON_CONTENT}" "pebble" "messageKeys")
    
    if(NOT JSON_ERR)
        set(HEADER_CONTENT "/* Auto-generated by CMake from package.json. Do not edit. */\n#pragma once\n\n")
        
        math(EXPR MAX_IDX "${MESSAGE_KEYS_LEN} - 1")
        foreach(IDX RANGE ${MAX_IDX})
            string(JSON KEY GET "${PACKAGE_JSON_CONTENT}" "pebble" "messageKeys" ${IDX})
            set(HEADER_CONTENT "${HEADER_CONTENT}#define MESSAGE_KEY_${KEY} 100${IDX}\n")
        endforeach()
        
        # Write the auto-generated header
        file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/generated/message_keys.auto.h" "${HEADER_CONTENT}")
        message(STATUS "Generated message_keys.auto.h with ${MESSAGE_KEYS_LEN} keys")
    else()
        message(WARNING "Could not parse messageKeys from package.json: ${JSON_ERR}")
    endif()
endif()

# Preprocessor definitions for Pebble SDK
add_definitions(
    -DPBL_COLOR=1
    -DPBL_COMPASS=1
    -DPBL_DISPLAY_HEIGHT=168
    -DPBL_DISPLAY_WIDTH=144
    -DPBL_HEALTH=1
    -DPBL_MICROPHONE=1
    -DPBL_PLATFORM_basalt=1
    -DPBL_RECT=1
)

    "/Users/jonny/Library/Application Support/Pebble SDK/SDKs/4.9.169/sdk-core/pebble/emery/include"
    "/Users/jonny/Library/Application Support/Pebble SDK/SDKs/4.9.169/sdk-core/pebble/gabbro/include"
    "src/c"
)

# Source files
set(SOURCES
    src/c/main.c
    src/c/navigation.c
)

add_executable(pebble-navi ${SOURCES})
